虽然估计对你没啥参考价值，贴出来吧……= =：[code]# -*- Makefile -*-
#
# Makefile.mingw32 - Makefile rules for MinGW
#

INCS = -I./dep/mingw32/include -I./dep/mingw32/include/SDL -I./dep/mingw32/include/smpeg
LIBS = -L./dep/mingw32/lib -lSDL_ttf -lmingw32 `pkg-config --static --libs SDL_image` `pkg-config --static --libs SDL_mixer` -lsmpeg -lSDL -lbz2 -ljpeg -lm -lfreetype -lvorbisfile -lvorbis -logg -lkernel32 -lgdi32 -luser32 -lwinmm -luuid -ldxguid -lSDLmain -lSDL
DEFS = -DWIN32 -D_WIN32 -DUSE_OGG_VORBIS -DENABLE_1BYTE_CHAR -DPTW32_STATIC_LIB
CHARSET=gbk

EXESUFFIX = .exe
OBJSUFFIX = .o

.SUFFIXES:
.SUFFIXES: $(OBJSUFFIX) .cpp .h

CC = i586-mingw32msvc-g++
LD = i586-mingw32msvc-g++ --static -o
CP = cp
RM = rm -f

CFLAGS = -O3 -Wall -fomit-frame-pointer -pipe -c $(INCS) $(DEFS)

TARGET = onscripter$(EXESUFFIX) sardec$(EXESUFFIX) nsadec$(EXESUFFIX) sarconv$(EXESUFFIX) nsaconv$(EXESUFFIX)

include Makefile.onscripter
