while true do

-- clear screen
Gu.clearDepth(255);
Gu.clear(Gu.COLOR_BUFFER_BIT+Gu.DEPTH_BUFFER_BIT)

-- setup projection and view matrices

Gum.matrixMode(Gu.PROJECTION)
Gum.loadIdentity()

Gum.perspective(75, 480/272, 0.5, 1000)--立方体(75越小,立方体越小)

Gum.matrixMode(Gu.VIEW)
Gum.loadIdentity()

---- setup matrix(矩阵) for triangle

---- setup texture

Gu.blendFunc(Gu.ADD, Gu.SRC_ALPHA, Gu.ONE_MINUS_SRC_ALPHA, 0, 0)

Gu.texImage(logo)
Gu.texFunc(Gu.TFX_MODULATE, Gu.TCC_RGBA)

Gu.texFilter(Gu.LINEAR, Gu.LINEAR)--Filter:过滤器,LINEAR:线的
Gu.texScale(5, 5)--Scale:尺度(0.5为一个图单位)
Gu.texOffset(0, 0)
Gu.ambientColor(white)--周围的颜色

---- setup matrix for cube

Gum.matrixMode(Gu.MODEL)
Gum.loadIdentity()
Gum.translate(x, y, -m);--(立方体中心坐标)

---- draw cube

Gum.drawArray(Gu.TRIANGLES, Gu.TEXTURE_32BITF+Gu.COLOR_8888+Gu.VERTEX_32BITF+Gu.TRANSFORM_3D, cube)--(对比1 :平面和立体矩阵)
--- Gum.drawArray(Gu.TRIANGLES, Gu.COLOR_8888+Gu.VERTEX_32BITF+Gu.TRANSFORM_3D, plane)--(对比2 :[Gu.TEXTURE_32BITF] -- 32位色纹理)








---- setup texture
