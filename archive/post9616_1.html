是动画序列,大概原理就是把图片的每一帧放到一个数组里,然后"播放"那个数组

cartoon = {}
cartoon[1] = LoadImage("1.png")
cartoon[2] = LoadImage("2.png")
cartoon[3] = LoadImage("3.png")
cartoon[4] = LoadImage("4.png")
cartoon["size"] = 4
cartoon["count"] = 1
cartoon["timer"] = 0

while true do
DrawImage(cartoon[cartoon["count"]],0,0)
cartoon["timer"] = cartoon["timer"] + 1
if cartoon["timer"] > 4 then
cartoon["timer"] = 0
cartoon["count"] = cartoon["count"] + 1
if cartoon["count"] > cartoon["size"] then
cartoon["count"] = 1
end
end
end

这是伪代码,方法还有其它,具体得看一下luaplayer具体有哪些功能
