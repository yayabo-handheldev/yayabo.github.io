function buttonshin()
if findtest() == 1 then fgon(1);fgoff(2);
elseif findtest() == 2 then fgon(2);fgoff(1);
elseif findtest() ~= 1 and findtest() ~= 2 then fgon(2);fgon(1);
end
end
